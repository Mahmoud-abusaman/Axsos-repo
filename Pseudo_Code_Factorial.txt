START
  INPUT N
  IF n <0 
    THROW "error"
  ELSE IF N IS EQUAL TO 0 OR N IS EQUAL TO 1
    PRINT 1
  ELSE 
    I = 2
    RESULT=1
    WHILE I <= N
      SET RESULT = RESULT * I
      I = I + 1
    ENDLOOP
    OUTPUT "THE FACTORIAL OF N IS" + RESULT
  ENDIF
END

FUNCTION factorial (n)
  IF n < 0 THEN
    THROW "error"
  ELSE IF n < 2 THEN
    RETURN 1
  Else
    Return n* Factorial(n-1)
  ENDIF
END FUNCTION
